[docs]    def start(self):
        self.stream.set(cv2.CAP_PROP_POS_FRAMES, self.current_index + 1)
        self.thread = Thread(target=self.fill_buffer)
        self.thread.daemon = True
        self.state = 1
        self.thread.start()



[docs]    def fill_buffer(self):
        while True:
            if self.state <= 0:
                break
            if not self.buffer.full():
                ret, frame = self.stream.read()
                if ret:
                    frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                    self.buffer.put(frame)
            else:
                time.sleep(0.01)



[docs]    def stop(self):
        if self.thread:
            self.state = -1
            self.thread.join()
            with self.buffer.mutex:
                self.buffer.queue.clear()
        self.thread = None

[docs]    def next_frame(self) -> np.ndarray:
        if self.state == -1:
            return None
        elif self.state != 1:
            self.start()
        try:
            self.current_frame = self.buffer.get(timeout=0.5)
            self.current_index += 1
            return self.current_frame
        except Empty:
            self.stop()
            return None
